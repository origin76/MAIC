\begin{denotation}
    \item[CTDE] 集中式训练、分布式执行（Centralized Training and Decentralized Execution）
    \item[MAPPO] 多智能体近端策略优化（Multi-Agent Proximal Policy Optimization）
    \item[PPO] 近端策略优化（Proximal Policy Optimization）
    \item[SMAC] 星际争霸多智能体挑战基准（StarCraft Multi-Agent Challenge）
    \item[$h_i^t$] 智能体 $i$ 在时刻 $t$ 的局部隐藏状态
    \item[$m_i^t$] 智能体 $i$ 在时刻 $t$ 接收到的通信消息
    \item[$\alpha_{ij}^t$] 智能体 $i$ 在时刻 $t$ 对智能体 $j$ 的注意力权重
    \item[$\Delta_i^t$] 通信分支对局部动作 logits 的残差修正量
\end{denotation}
